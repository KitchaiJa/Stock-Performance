tickets <- c('KTB.BK', 'KBANK.BK', 'BBL.BK', 'BAY.BK')
getSymbols(tickets, src='yahoo', from=Sys.Date()-1095)
## [1] "KTB.BK"   "KBANK.BK" "BBL.BK"   "BAY.BK"
closing_price <- cbind(KTB.BK[,6], KBANK.BK[,6], BBL.BK[,6], BAY.BK[,6]) |> round(2)

return_price <- cbind(
  KTB <- weeklyReturn(KTB.BK, type = 'log'),
  KBANK <- weeklyReturn(KBANK.BK, type = 'log'),
  BBL <- weeklyReturn(BBL.BK, type = 'log'),
  BAY <- weeklyReturn(BAY.BK, type = 'log')
) |> round(2)


#change column name
colnames(return_price) <- c('KTB', 'KBANK', 'BBL', 'BAY')

#Calculate Value at Risk
CVaR(return_price, p=0.99) |> round(2)
##      KTB KBANK   BBL   BAY
## ES -0.16 -0.23 -0.17 -0.27
#plot
PerformanceAnalytics::charts.PerformanceSummary(return_price,
